\newglossaryentry{programa}{
    name={programa},
    description={Un programa informático es un conjunto de instrucciones que el indican al ordenador las operaciones que debe realizar para lograr un fin determinado}
}

\newglossaryentry{registros}{
    name={registros},
    description={Son una pequeña porción de memoria compuesta de biestables que permiten guardar valores en binario. Su tamaño viene dado por el número de bits que pueden almacenar; por ejemplo, puede ser un registro de 8 bits" o un de 32 bits}
}

\newglossaryentry{volatil}{
    name={volátil},
    description={Se conoce como memoria volátil aquella que pierde la información cuando se interrumpe el flujo eléctrico}
}

\newglossaryentry{interfaces}{
    name={interfaces},
    description={Una interfaz hardware es una conexión física, entre dos aparatos o sistemas independientes, que funciona a través de un protocolo común a ambos. Existen diferentes interfaces estándares como el USB o el SATA, etc., definidas con unas especificaciones técnicas concretas, que deben ser comunes en todos los dispositivos con la misma interfaz, para que puedan conectarse y comunicarse entre ellos}
}

\newglossaryentry{bahias}{
    name={bahías},
    description={En las cajas de los ordenadores, son los compartimentos destinados a albergar los dispositivos ópticos, los discos duros, entre otros posibles dispositivos}
}

\newglossaryentry{BIOS}{
    name={BIOS},
    description={Viene de las siglas Basic Input/Output System y consiste en un pequeño programa muy básico instalado en la placa base, con el que se pueden configurar y manejar el hardware imprescindible que permite poner en marcha el computador}
}

\newglossaryentry{SVGA}{
    name={SVGA},
    description={Super Video Graphic Array, también conocida como Super VGA o Dsub 15, es un tipo de conector con 15 dispuestos en tres filas que fue diseñado para la conexión de monitores CRT a la tarjeta gráfica y que aún se sigue usando en los actuales monitores planos}
}

\newglossaryentry{S-Video}{
    name={S-Video},
    description={Significa Vídeo por Separado y se trata de un conector con el que se puede conectar el ordenador al televisor. La señal de vídeo que envía es analógica y transmite por separado la información de brillo y de color, de ahí su nombre}
}

\newglossaryentry{DVI}{
    name={DVI},
    description={Digital Video Interface, en ingles o Interfaz Digital de Video, en castellano. Es un tipo de conector diseñado para su utilización con los nuevos monitores o proyectores digitales. Este estándar de visualización evita la perdida de calidad de imagen producida  por el ordenador al no tener que convertirla en analógica para reproducirla}
}

\newglossaryentry{HDMI}{
    name={HDMI},
    description={High-Definition Multimedia Interface o Interfaz Multimedia de Alta Definición. Permite el uso de vídeo digital de alta definición, así como de audio multicanal en un único cable}
}

\newglossaryentry{MIDI}{
    name={MIDI},
    description={Técnica de creación de música digital a partir de tablas de muestras de sonidos previamente almacenados, que se reproducen cuando se especifica en la partitura}
}
